1. Canvas Kya Hai?

<canvas> ek drawing surface hai — JavaScript se shapes, text, images, aur animations draw kar sakte hain.

<canvas id="myCanvas" width="400" height="300">
  Aapka browser Canvas support nahi karta.
</canvas>

<script>
  let canvas = document.getElementById("myCanvas");
  let ctx = canvas.getContext("2d");   // 2D drawing context
  // Ab ctx se drawing karein
</script>

2. Rectangles Draw Karna

ctx.fillStyle = "#e94560";            // Fill color
ctx.fillRect(50, 50, 200, 100);       // x, y, width, height

ctx.strokeStyle = "#0f3460";          // Border color
ctx.lineWidth = 3;
ctx.strokeRect(50, 50, 200, 100);     // Border rectangle

ctx.clearRect(80, 70, 50, 50);        // Area clear/erase karna

3. Lines Draw Karna

ctx.beginPath();           // Naya path start
ctx.moveTo(50, 50);        // Starting point
ctx.lineTo(200, 50);       // Line draw to
ctx.lineTo(200, 150);      // Aur ek line
ctx.strokeStyle = "#00d2ff";
ctx.lineWidth = 2;
ctx.stroke();              // Lines visible karna

4. Circles aur Arcs

// arc(centerX, centerY, radius, startAngle, endAngle, anticlockwise)
ctx.beginPath();
ctx.arc(200, 100, 60, 0, Math.PI * 2);  // Full circle
ctx.fillStyle = "#0f3460";
ctx.fill();

// Half circle
ctx.beginPath();
ctx.arc(200, 100, 60, 0, Math.PI);      // 180 degrees = half
ctx.stroke();

5. Text Draw Karna

ctx.font = "30px Arial";
ctx.fillStyle = "#e94560";
ctx.fillText("Code Academy!", 50, 60);     // Filled text

ctx.strokeStyle = "#0f3460";
ctx.lineWidth = 1;
ctx.strokeText("Nowshera", 50, 110);        // Outlined text

ctx.textAlign = "center";                   // center, left, right

6. Gradients

// Linear Gradient
let grad = ctx.createLinearGradient(0, 0, 400, 0);
grad.addColorStop(0, "#e94560");
grad.addColorStop(1, "#0f3460");
ctx.fillStyle = grad;
ctx.fillRect(0, 0, 400, 100);
